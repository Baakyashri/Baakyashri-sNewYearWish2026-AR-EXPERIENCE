    const starsContainer = document.getElementById('stars');
    for (let i = 0; i < 100; i++) {
      const star = document.createElement('div');
      star.className = 'star';
      star.style.left = Math.random() * 100 + '%';
      star.style.top = Math.random() * 100 + '%';
      star.style.animationDelay = Math.random() * 3 + 's';
      starsContainer.appendChild(star);
    }

    function createConfetti() {
      for (let i = 0; i < 50; i++) {
        setTimeout(function() {
          const confetti = document.createElement('div');
          confetti.className = 'confetti';
          confetti.style.left = Math.random() * 100 + '%';
          const colors = ['#ff0844', '#ffb199', '#ffd93d', '#6bcf7f', '#4d96ff'];
          confetti.style.backgroundColor = colors[Math.floor(Math.random() * 5)];
          document.body.appendChild(confetti);
          setTimeout(function() {
            confetti.remove();
          }, 3000);
        }, i * 30);
      }
    }

    const arButton = document.getElementById('ar-button');
    const modelViewer = document.getElementById('cake-model');
    
    arButton.addEventListener('click', function() {
      createConfetti();
    });

    setTimeout(function() {
      if (modelViewer.canActivateAR === false) {
        const deviceNote = document.querySelector('.device-note');

        deviceNote.style.background = 'rgba(255,100,100,0.3)';
        deviceNote.style.borderColor = 'rgba(255,100,100,0.5)';
      }
    }, 1000);

    modelViewer.addEventListener('ar-status', function(event) {
